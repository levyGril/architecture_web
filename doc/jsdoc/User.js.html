/**
 * User实体类
 * @class
 */
 class User{

    /**
     *
     * @param {string} name - 姓名
     * @param {number} age - 年龄
     * @constructor
     */
    constructor(name,age){
        this.name = name;
        this.age = age;
    }

    /**
     * 获取数据的api
     * @method getUserData
     * @returns {Promise} -- 返回异步数据处理结果
     * @function
     */
    getUserData(){
        return new Promise((resolve,reject)=>{
            setTimeout(()=>{
                resolve("我来自User model的getUserData方法,我的名字是： "+this.name +", 我今年 "+this.age+"了！！！");
            },1000);
        })
    }
}

export default User;
